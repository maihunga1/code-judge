// Function to parse the saved test cases from a file
func parseTestCases(fileContent string) []struct {
	nums   []int
	target int
	result []int
} {
	testCases := strings.Split(strings.TrimSpace(fileContent), "\n\n") // Split into individual test cases

	var parsedCases []struct {
		nums   []int
		target int
		result []int
	}

	for _, testCase := range testCases {
		lines := strings.Split(strings.TrimSpace(testCase), "\n")

		// Parse nums
		numStrs := strings.Split(strings.TrimSpace(lines[0]), " ")
		var nums []int
		for _, numStr := range numStrs {
			num, _ := strconv.Atoi(strings.TrimSpace(numStr))
			nums = append(nums, num)
		}

		// Parse target
		target, _ := strconv.Atoi(strings.TrimSpace(lines[1]))

		// Parse expected result
		var result []int
		if len(lines) > 2 && strings.TrimSpace(lines[2]) != "" {
			resultStrs := strings.Split(strings.TrimSpace(lines[2]), ", ")
			for _, resStr := range resultStrs {
				res, _ := strconv.Atoi(strings.TrimSpace(resStr))
				result = append(result, res)
			}
		}

		parsedCases = append(parsedCases, struct {
			nums   []int
			target int
			result []int
		}{
			nums:   nums,
			target: target,
			result: result,
		})
	}

	return parsedCases
}

// Helper function to compare two arrays for equality
func arraysAreEqual(arr1, arr2 []int) bool {
	if len(arr1) != len(arr2) {
		return false
	}
	for i := range arr1 {
		if arr1[i] != arr2[i] {
			return false
		}
	}
	return true
}

// Function to read and test the cases from the file
func readAndTestCases(txtFilename string) {
	// Open the file
	file, err := os.Open(txtFilename)
	if err != nil {
		fmt.Printf("Error opening file: %v\n", err)
		return
	}
	defer file.Close()

	// Read the entire file content
	var fileContent string
	scanner := bufio.NewScanner(file)
	for scanner.Scan() {
		fileContent += scanner.Text() + "\n"
	}

	if err := scanner.Err(); err != nil {
		fmt.Printf("Error reading file: %v\n", err)
		return
	}

	// Parse the test cases
	testCases := parseTestCases(fileContent)

	// Run the test cases
	for index, testCase := range testCases {
		output := twoSum(testCase.nums, testCase.target)

		if !arraysAreEqual(output, testCase.result) {
			fmt.Printf("Test case %d failed:\n", index+1)
			fmt.Printf("  Input nums: %v\n", testCase.nums)
			fmt.Printf("  Target: %d\n", testCase.target)
			fmt.Printf("  Expected: %v\n", testCase.result)
			fmt.Printf("  Got: %v\n", output)
			return
		}
	}

	fmt.Println("All test cases passed.")
}

func main() {
	// Example usage: run tests from the file
	readAndTestCases("test-case.txt")
}
